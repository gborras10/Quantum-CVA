import numpy as np
import matplotlib.pyplot as plt

# ============================================================
# 1) Construcción de métricas de error relevantes para el CVA
# ============================================================

M = len(default_probabilities_target)
N = len(qcbm_target_distribution) // M

p_joint = qcbm_probabilities_statevector.reshape(M, N)
v_pred_2d = positive_exposure_statevector.reshape(M, N)
v_tgt_2d = positive_exposure_target.reshape(M, N)

p_tilde = discount_factors_statevector.reshape(M, 1)
q_tilde = default_probabilities_statevector.reshape(M, 1)

# Error bruto del bloque exposure
abs_v_err = np.abs(v_pred_2d - v_tgt_2d)
signed_v_err = v_pred_2d - v_tgt_2d

# Peso real de cada estado dentro del CVA
state_weight = (
    M
    * (1.0 - recovery_rate)
    * C_v
    * C_p
    * C_q
    * p_joint
    * p_tilde
    * q_tilde
)

# Contribución por estado al error del CVA debida SOLO a exposure
signed_cva_err = state_weight * signed_v_err
abs_cva_err = np.abs(signed_cva_err)

# Resúmenes globales
total_signed_exposure_cva_error = float(np.sum(signed_cva_err))
total_abs_exposure_cva_error = float(np.sum(abs_cva_err))

print("=== Exposure -> CVA error summary ===")
print("Total signed CVA error from exposure only =", total_signed_exposure_cva_error)
print("Total absolute CVA error from exposure only =", total_abs_exposure_cva_error)

# ============================================================
# 2) Ranking de estados por contribución REAL al error del CVA
# ============================================================

flat_abs_v_err = abs_v_err.ravel()
flat_state_weight = state_weight.ravel()
flat_abs_cva_err = abs_cva_err.ravel()
flat_signed_cva_err = signed_cva_err.ravel()
flat_p_joint = p_joint.ravel()
flat_v_pred = v_pred_2d.ravel()
flat_v_tgt = v_tgt_2d.ravel()

idx_sorted = np.argsort(-flat_abs_cva_err)

cum_abs_cva = np.cumsum(flat_abs_cva_err[idx_sorted])
cum_abs_cva_share = cum_abs_cva / max(cum_abs_cva[-1], 1e-16)

print("\n=== Top 15 states by absolute contribution to CVA error ===")
for rank, idx in enumerate(idx_sorted[:15], start=1):
    i = idx // N
    j = idx % N
    print(
        f"{rank:2d}. state (t={i}, s={j}) | "
        f"P={flat_p_joint[idx]:.6e} | "
        f"weight={flat_state_weight[idx]:.6e} | "
        f"v_pred={flat_v_pred[idx]:.6e} | "
        f"v_tgt={flat_v_tgt[idx]:.6e} | "
        f"|dv|={flat_abs_v_err[idx]:.6e} | "
        f"signed dCVA={flat_signed_cva_err[idx]:.6e} | "
        f"|dCVA|={flat_abs_cva_err[idx]:.6e}"
    )

for k in [1, 3, 5, 10, 15, 20]:
    share = cum_abs_cva_share[min(k, len(cum_abs_cva)) - 1]
    print(f"Top {k:2d} states explain {100 * share:.2f}% of total |dCVA| from exposure")

# ============================================================
# 3) Gráficas
# ============================================================

eps = 1e-16

fig, axes = plt.subplots(2, 2, figsize=(14, 10))

# (a) Heatmap del error bruto de exposure
im0 = axes[0, 0].imshow(abs_v_err, aspect="auto")
axes[0, 0].set_title(r"Error bruto de exposure: $|v_{\mathrm{pred}}-v_{\mathrm{tgt}}|$")
axes[0, 0].set_xlabel("price-bin index")
axes[0, 0].set_ylabel("time index")
fig.colorbar(im0, ax=axes[0, 0])

# (b) Heatmap de contribución absoluta al error del CVA
im1 = axes[0, 1].imshow(abs_cva_err, aspect="auto")
axes[0, 1].set_title(r"Contribución absoluta al error del CVA: $|\Delta \mathrm{CVA}_{ij}^{(v)}|$")
axes[0, 1].set_xlabel("price-bin index")
axes[0, 1].set_ylabel("time index")
fig.colorbar(im1, ax=axes[0, 1])

# (c) Scatter: error bruto vs peso real
axes[1, 0].scatter(flat_abs_v_err + eps, flat_state_weight + eps, s=20)
axes[1, 0].set_xscale("log")
axes[1, 0].set_yscale("log")
axes[1, 0].set_xlabel(r"$|v_{\mathrm{pred}}-v_{\mathrm{tgt}}|$")
axes[1, 0].set_ylabel(r"peso real en CVA")
axes[1, 0].set_title("Estados con gran error bruto pero poco peso real quedan abajo")

# Marcar top 10 por contribución al error del CVA
for idx in idx_sorted[:10]:
    i = idx // N
    j = idx % N
    axes[1, 0].annotate(
        f"({i},{j})",
        (flat_abs_v_err[idx] + eps, flat_state_weight[idx] + eps),
        fontsize=8,
    )

# (d) Pareto acumulado de contribución absoluta
axes[1, 1].plot(np.arange(1, len(cum_abs_cva_share) + 1), cum_abs_cva_share)
axes[1, 1].set_xlabel("número de estados (ordenados por |dCVA|)")
axes[1, 1].set_ylabel("fracción acumulada de |dCVA| total")
axes[1, 1].set_title("Concentración del error relevante")

plt.tight_layout()
plt.show()

# ============================================================
# 4) Figura extra: comparación directa entre error bruto y error relevante
# ============================================================

fig, axes = plt.subplots(1, 2, figsize=(14, 4))

# Ordenados por error bruto
idx_err = np.argsort(-flat_abs_v_err)
axes[0].plot(flat_abs_v_err[idx_err], label="|dv|")
axes[0].plot(flat_abs_cva_err[idx_err], label="|dCVA| por estado")
axes[0].set_yscale("log")
axes[0].set_title("Ordenando por error bruto de exposure")
axes[0].set_xlabel("estado ordenado")
axes[0].legend()

# Ordenados por contribución real al CVA
idx_cva = np.argsort(-flat_abs_cva_err)
axes[1].plot(flat_abs_v_err[idx_cva], label="|dv|")
axes[1].plot(flat_abs_cva_err[idx_cva], label="|dCVA| por estado")
axes[1].set_yscale("log")
axes[1].set_title("Ordenando por contribución real al error del CVA")
axes[1].set_xlabel("estado ordenado")
axes[1].legend()

plt.tight_layout()
plt.show()

# Estados con error bruto grande pero contribución pequeña
thr_err = np.quantile(flat_abs_v_err, 0.90)       # top 10% en |dv|
thr_cva = np.quantile(flat_abs_cva_err, 0.50)     # mitad inferior en |dCVA|

mask = (flat_abs_v_err >= thr_err) & (flat_abs_cva_err <= thr_cva)

print("\n=== States with large raw exposure error but small CVA relevance ===")
for idx in np.where(mask)[0][:20]:
    i = idx // N
    j = idx % N
    print(
        f"(t={i}, s={j}) | "
        f"P={flat_p_joint[idx]:.6e} | "
        f"weight={flat_state_weight[idx]:.6e} | "
        f"|dv|={flat_abs_v_err[idx]:.6e} | "
        f"|dCVA|={flat_abs_cva_err[idx]:.6e}"
    )
